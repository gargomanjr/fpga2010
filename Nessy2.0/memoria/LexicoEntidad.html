222     /**
223      * Para saber si un caracter determinado es una letra
224      * @param ch El caracter del que se quiere saber si es una letra
225      * @return true si ch es una letra y false en caso contrario
226      */
227     private boolean esLetra(Character ch) {
228         return (ch >= 'a' && ch <= 'z') || (ch >= 'A' && ch <= 'Z');
229     }

231     /**
232      * Para saber si un caracter determinado es un dígito
233      * @param ch El caracter del que se quiere saber si es un dígito
234      * @return true si ch es un dígito y false en caso contrario
235      */
236     private boolean esDigito(Character ch) {
237         return ch >= '0' && ch <= '9';
238     }

240     /**
241      * Lee un caracter del fichero y lo devuelve
242      * @return El caracter leído o null en caso de que no se haya podido leer
243      * @throws IOException
244      */
245     private Character leerCaracter() throws IOException {
246         if (reader == null) {
247             errores.error("No se asigno ningún reader");
248             throw new IOException("No se asigno ningún reader");
249         }
250         if (reader.ready()) {
251             int ch = reader.read();
252             if (ch != -1) //si no es fin de fichero
253             {
254                 ultimoCharLeido = new Character((char) ch);
255             } else {
256                 ultimoCharLeido = null;
257             }
258 
259             if (ultimoCharLeido.charValue() == '\n') { //Si es un salto de lÃ­nea
260                 numLinea++; //incrementa la linea
261             } 
262             return ultimoCharLeido;
263         } else {
264             return null;
265         }
266     }

268     /**
269      * Método que gestiona el autómata finito que ocntrola el Analizador Léxico
270      * @return El siguiente Token a leer.
271      * @throws IOException
272      */
273     public Token sigToken() throws IOException {
274         estado = 0;
275         cadena = "";
276         while (true) {
277             char caracterLeido = 0;
278             if (ultimoCharLeido != null) {
279                 caracterLeido = Character.toLowerCase(ultimoCharLeido);
280             }
281             switch (estado) {
282                 case 0:
283                     if (this.ultimoCharLeido == null) {
284                         transita(6);
285 
286                     } else if ((ultimoCharLeido.charValue() == ' ') || (ultimoCharLeido.charValue() == '\n') || (ultimoCharLeido.charValue() == '\r') || (ultimoCharLeido.charValue() == '\t')) {
287                         transita(0); //salta los blancos
288                         cadena = "";
289                     } else if (esLetra(caracterLeido)) {
290                         transita(1);
291                     } else if (esDigito(caracterLeido)) {
292                         transita(2);
293                     } else if (caracterLeido == ';') {
294                         transita(3);
295                     } else if (caracterLeido == '(') {
296                         transita(4);
297                     } else if (caracterLeido == ')') {
298                         transita(5);
299                     } else if (caracterLeido == ':') {
300                         transita(7);
301                     } else if (caracterLeido == '-') {//posible comentario
302                         transita(8);
303                     } else if (caracterLeido == '+') {
304                         transita(12);
305                     } else if (caracterLeido == '*') {
306                         transita(13);
307                     }else if (caracterLeido == '/') {
308                         transita(14);
309                     } else {//caracter desconocido
310                         transita(10);
311                     }
312                     break;
313                 case 1:
314                     if (esLetra(caracterLeido) || esDigito(caracterLeido) || caracterLeido == '_') {
315                         transita(1);
316                     } else {
317                         Integer codigo = palabrasReservadas.get(cadena.toUpperCase());
318                         if (codigo != null) {//si es una palabra reservada
319                             return new Token(codigo, cadena, numLinea);
320                         } else {//si es el nombre de una entrada/salida/entidad
321                             return new Token(IDENTIFICADOR, cadena, numLinea);
322                         }
323                     }
324                     break;
325                 case 2:
326                     if (esDigito(caracterLeido)) {//continua el número
327                         transita(2);
328                     } else {//guardar el numero
329                         return new Token(ENTERO, cadena, numLinea);
330                     }
331                     break;
332                 case 3:
333                     return new Token(PUNTO_Y_COMA, cadena, numLinea);
334                 case 4:
335                     return new Token(ABRE_PARENTESIS, cadena, numLinea);
336                 case 5:
337                     return new Token(CIERRA_PARENTESIS, cadena, numLinea);
338                 case 6:
339                     return new Token(EOF, cadena, numLinea);
340                 case 7:
341                     if (caracterLeido == '=') {
342                         transita(11);
343                     } else {
344                         return new Token(DOS_PUNTOS, cadena, numLinea);
345                     }
346                     break;
347                 case 8:
348                     if (caracterLeido == '-') {
349                         transita(9);
350                     } else {
351                         return new Token(RESTA, cadena, numLinea);
352                     }
353                     break;
354                 case 9://comentarios
355                     if (caracterLeido == '\n' || caracterLeido == '\r') {
356                         transita(0);
357                     } else {
358                         transita(9);
359                     }
360                     break;
361                 case 10://desconocido
362                     return new Token(OTRO, cadena, numLinea);
363                 case 11:
364                     return new Token(ASIG_GENERIC, cadena, numLinea);
365                 case 12:
366                     return new Token(SUMA, cadena, numLinea);
367                 case 13:
368                     return new Token(MULT, cadena, numLinea);
369                 case 14:
370                     return new Token(DIV, cadena, numLinea);
371             }
372         }
373     }

375     /**
376      * Cambia el estado actual del analizador léxico por el que se le pasa por 
377      * parámetro. Además lee el siguiente caracter para dejarlo preparado para
378      * ser analizado.
379      * @param sigEstado Siguiente estado al que va el analizador.
380      * @throws IOException
381      */
382     public void transita(int sigEstado) throws IOException {
383         estado = sigEstado;
384         if (ultimoCharLeido != null) {
385             cadena = cadena.concat(ultimoCharLeido.toString());
386             ultimoCharLeido = leerCaracter();
387         }
388     }
